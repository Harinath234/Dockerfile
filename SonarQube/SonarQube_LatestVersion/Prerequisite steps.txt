#Install Docker
##Create Project Directory
mkdir sonarqube-docker
cd sonarqube-docker

##Download SonarQube Manually
wget https://binaries.sonarsource.com/Distribution/sonarqube/sonarqube-26.1.0.118079.zip

##Extract:
sudo apt update
sudo apt install -y unzip

unzip sonarqube-26.1.0.118079.zip
mv sonarqube-26.1.0.118079 sonarqube

vi Dockerfile
